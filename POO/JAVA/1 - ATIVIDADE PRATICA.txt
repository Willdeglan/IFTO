package javaapplication1;

import java.util.Scanner;

public class at1 {

    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        double b1, b2, h, result;
        
        System.out.println("Vamos calcular a área do trapézio.");
        System.out.println("Diga o valor para a Base Maio em metro: ");
            b1 = sc.nextDouble();
    
        System.out.println("Diga o valor para a Base Menor em metros: ");
            b2 = sc.nextDouble();
       
        System.out.println("Diga o valor para a Altura em metros: ");
            h = sc.nextDouble();
    
        result = (b1 + b2)*h/2;    
        
        System.out.println("A área do trapezio é: "+result+"m²");
    }
    
}

    public static void main(String[] args) {
        
        Scanner sc = new Scanner(System.in);
        
        double pes, imc=0;
        float alt;
        
        System.out.println("Vamos calcular o IMC de uma pessoa? ");
        System.out.println("Qual o peso em quilos? ");
            pes = sc.nextDouble();
        
        System.out.println("Qual a altura em centimetros? ");
            alt = sc.nextFloat();
            
        imc = pes/(alt*alt);
        
        System.out.println("O IMC é: "+imc*10000+"K/m²");
    }
    
}
